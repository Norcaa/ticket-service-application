    @Override
    public boolean equals(Object o) {
        if (this == o) {
            return true;
        }
        if (o == null || getClass() != o.getClass()) {
            return false;
        }
        Room that = (Room) o;
        return that.rows.equals(rows) && that.columns.equals(columns) && Objects.equals(name, that.name);
    }

    @Override
